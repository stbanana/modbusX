cmake_minimum_required(VERSION 3.22)
project(ARRAY_TEST)

# gtest库相关
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../../googletest)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/googletest/include)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/googlemock/include)

find_library(GTEST_LIB gtest HINTS ${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/build/lib)
find_library(GTEST_MAIN_LIB gtest_main HINTS ${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/build/lib)
find_library(GMOCK_LIB gmock HINTS ${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/build/lib)
find_library(GMOCK_MAIN_LIB gmock_main HINTS ${CMAKE_CURRENT_SOURCE_DIR}/../../googletest/build/lib)

# 确保路径正确性
set(MBX_COMMON_INC ${CMAKE_CURRENT_SOURCE_DIR}/../../../common)

# 包含 common 库的头文件目录
include_directories(${MBX_COMMON_INC}/include)
include_directories(${MBX_COMMON_INC}/../port/generic/inc)

# 递归查找所有 common 库的源码文件
file(GLOB_RECURSE MBX_COMMON_SRC ${MBX_COMMON_INC}/source/*.c)

# 递归查找所有当前测试用例的源码文件
file(GLOB_RECURSE SRC_C ${CMAKE_CURRENT_LIST_DIR}/*.c)
file(GLOB_RECURSE SRC_CPP ${CMAKE_CURRENT_LIST_DIR}/*.cpp)
set(SRC ${SRC_C} ${SRC_CPP})

# 生成测试可执行程序
set(BUILD_PROJECT_NAME test_library)
add_executable(${BUILD_PROJECT_NAME})
target_compile_definitions(${BUILD_PROJECT_NAME} PRIVATE)
target_include_directories(${BUILD_PROJECT_NAME} PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}
)

target_sources(${BUILD_PROJECT_NAME} PRIVATE
    ${SRC}
    ${MBX_COMMON_SRC}
)

# 链接 gtest 和 gmock 库，并显式链接 C++ 标准库
target_link_libraries(${BUILD_PROJECT_NAME} PRIVATE
    ${GTEST_LIB}      
    ${GTEST_MAIN_LIB} 
    ${GMOCK_LIB}
    ${GMOCK_MAIN_LIB}
    pthread
)

# 添加测试
add_test(NAME ${BUILD_PROJECT_NAME} COMMAND ${BUILD_PROJECT_NAME})
